# P7-P7-VE31: displacement_map
## Mission Context
P7-VE31: Displacement Map Effect - Porting legacy shader to VJLive3 plugin architecture
## Technical Requirements
- GPU-accelerated displacement mapping using FBM noise
- 21 parameters with precise remapping from UI 0-10 range
- Multiple displacement sources: self-luminance, self-RG, noise, radial, edge, curl flow
- Channel mapping: RG→XY, luma→radial, luma→twist, luma→zoom
- Wrap modes: clamp, repeat, mirror, zoom_edge, black
- Chromatic displacement (RGB channels displaced differently)
- Anti-aliasing via multi-sampling
- Edge darkening based on displacement magnitude
- Feedback iteration for recursive displacement
- Sine wave modulation of displacement
- Safety rails: 60 FPS, ≥80% test coverage, ≤750 lines, no silent failures

    def __init__(self, strength=3.0, strength_x=5.0, strength_y=5.0, source_mode=0.0, channel_mode=0.0, bias=5.0, noise_scale=3.0, noise_speed=2.0, noise_octaves=3.0, noise_lacunarity=5.0, modulate_freq=0.0, modulate_amount=0.0, modulate_phase=0.0, feedback=0.0, wrap_mode=0.0, antialias=0.0, chromatic_displace=0.0, edge_darken=0.0, distort_rotate=0.0, pixelate=0.0):
        super().__init__()
        self.parameters = {
            # Core
            "strength": 3.0,           # Moderate displacement
            "strength_x": 5.0,         # Equal X
            "strength_y": 5.0,         # Equal Y
            "source_mode": 0.0,        # Self-luminance
            "channel_mode": 0.0,       # RG→XY
            "bias": 5.0,               # Neutral bias
            # Noise
            "noise_scale": 3.0,        # Medium scale
            "noise_speed": 2.0,        # Slow
            "noise_octaves": 3.0,      # Medium complexity
            "noise_lacunarity": 5.0,   # Standard
            # Modulation
            "modulate_freq": 0.0,      # No modulation
            "modulate_amount": 0.0,    # No modulation
            "modulate_phase": 0.0,     # Zero phase
            "feedback": 0.0,           # No feedback
            # Rendering
            "wrap_mode": 0.0,          # Clamp
            "antialias": 0.0,          # No AA
            "chromatic_displace": 0.0, # No chromatic
            "edge_darken": 0.0,        # No dark edges
            "distort_rotate": 0.0,     # No rotation
            "pixelate": 0.0,          # No pixelation
        }
        self.shader_code = """
        // FBM noise generation
        float fbm(vec2 pos) {
            float value = 0.0;
            float amplitude = 0.5;
            float frequency = 1.0;
            for (int i = 0; i < 4; i++) {
                value += amplitude * noise(pos * frequency);
                frequency *= 2.0;
                amplitude *= 0.5;
            }
            return value;
        }
        // Displacement calculation
        vec2 displacement(vec2 uv) {
            vec2 displacement = vec2(0.0);
            // Source modes
            if (source_mode == 0.0) { // Self-luminance
                displacement = vec2(luminance(uv));
            } else if (source_mode == 1.0) { // Self-RG
                displacement = vec2(uv.r - uv.g);
            } else if (source_mode == 2.0) { // Noise
                displacement = fbm(uv) * noise_scale;
            } else if (source_mode == 3.0) { // Radial
                displacement = fbm(uv) * noise_scale * radial(uv);
            } else if (source_mode == 4.0) { // Edge
                displacement = fbm(uv) * noise_scale * edge(uv);
            } else if (source_mode == 5.0) { // Curl flow
                displacement = fbm(uv) * noise_scale * curl_flow(uv);
            }
            // Channel mapping
            if (channel_mode == 0.0) { // RG→XY
                displacement = vec2(displacement.r, displacement.g);
            } else if (channel_mode == 1.0) { // luma→radial
                displacement = vec2(luminance(uv), displacement.r);
            } else if (channel_mode == 2.0) { // luma→twist
                displacement = vec2(luminance(uv) * cos(displacement.r), luminance(uv) * sin(displacement.r));
            } else if (channel_mode == 3.0) { // luma→zoom
                displacement = vec2(luminance(uv) * zoom(uv), displacement.r);
            }
            // Remapping
            displacement *= strength;
            displacement.x *= strength_x;
            displacement.y *= strength_y;
            // Feedback
            displacement += feedback * displacement;
            // Modulation
            displacement += modulate_freq * sin(displacement * modulate_phase);
            return displacement;
        }
        // Wrap modes
        vec2 wrap(vec2 uv, int mode) {
            if (mode == 0) return clamp(uv, 0.0, 1.0); // Clamp
            else if (mode == 1) return repeat(uv, 0.0, 1.0); // Repeat
            else if (mode == 2) return mirror(uv, 0.0, 1.0); // Mirror
            else if (mode == 3) return zoom_edge(uv, 0.0, 1.0); // Zoom edge
            else return black(uv); // Black
        }
        // Chromatic displacement
        vec3 chromatic_displacement(vec3 color, vec2 uv) {
            vec2 displacement = displacement(uv);
            return color + vec3(displacement.x, displacement.y, 0.0) * chromatic_displace;
        }
        // Anti-aliasing
        vec4 anti_aliased(vec2 uv) {
            return mix(texture2D(input, uv), texture2D(input, uv + vec2(0.5, 0.0)), antialias);
        }
        // Edge darkening
        float edge_darkening(vec2 uv) {
            return 1.0 - smoothstep(0.0, 0.1, abs(displacement(uv).x));
        }
        // Main render function
        vec4 render(vec2 uv) {
            vec2 displacement = displacement(uv);
            vec3 color = texture2D(input, uv + displacement).rgb;
            color = chromatic_displacement(color, uv);
            color = anti_aliased(uv).rgb;
            color *= edge_darkening(uv);
            return vec4(color, 1.0);
        }
        ""
    }
